def find_sum_of_distinct_elements(set1, set2): distinct_elements = set() total_sum = 0 for element in set1: distinct_elements.add(element) for element in set2: distinct_elements.add(element) for element in distinct_elements: total_sum += element return total_sum # Example usage set1 = {1, 2, 3, 4} set2 = {3, 4, 5, 6} result = find_sum_of_distinct_elements(set1, set2) print("Sum of distinct elements:", result)
